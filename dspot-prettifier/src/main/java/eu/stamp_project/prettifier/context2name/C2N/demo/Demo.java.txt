import spoon.Launcher;

public class Demo {
    private static String str = "ss";
    private final String mess = "mess-Demo";
    private final String global = "Global";

    private void print(int id) {
        String mess = "mess-print";
        System.out.println(mess);
        String local = "Local" + id;
        System.out.println(global + local);
    }

    private void label() {
        String mess = "mess-label";
        System.out.println(mess);
        outer:
        for (int i = 0; i < 10; i++) {
            inner:
            for (int j = 10; j > 0; j--) {
                if (i != j) {
                    System.out.println("break as i" + i + "j" + j);
                    break outer;
                } else {
                    System.out.println("continue as i" + i + "j" + j);
                    continue inner;
                }
            }
        }
    }

    private void exception() {
        try {
            throw Exception;
        } catch (Exception ex) {
            ex.printStackTrace();
        }
    }

    public static void main(String[] args) {
        new Demo().print(9);
        System.out.println(Demo.str);
    }
}
