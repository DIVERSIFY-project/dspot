import spoon.Launcher;

public class Demo {

    private static String str = "ss";

    private final String x = "mess-Demo";

    private final String type = "Global";

    private void print(int node) {
        String x = "mess-print";
        System.out.println(x);
        String key = "Local" + node;
        System.out.println(type + key);
    }

    private void label() {
        String x = "mess-label";
        System.out.println(x);
        ex: for (int value = 0; value < 10; value++) {
            context: for (int c = 10; c > 0; c--) {
                if (value != c) {
                    System.out.println("break as i" + value + "j" + c);
                    break ex;
                } else {
                    System.out.println("continue as i" + value + "j" + c);
                    break context;
                }
            }
        }
    }

    private void exception() {
        try {
            throw Exception;
        } catch (Exception name) {
            name.printStackTrace();
        }
    }

    public static void main(String[] e) {
        new Demo().print(9);
        System.out.println(Demo.str);
    }
}
